\section{Research Projects}
\cventry{\footnotesize{2016}}{Content Enhanced Network Embedding (CENE)}{}{}{[\href{https://github.com/Adoni/CENE}{\textcolor{color1}{Code}}]}{
\begin{itemize}
    \item {A novel approaches for learning network embeddings that could jointly leverage structure and content information in a social network.}
    \item {Applications on node classification (attributes: gender, profession, location) showed that our proposed approach significantly improved the F1 score up to 7\% when comparing to existing network embedding models.}
\end{itemize}}
\cventry{\footnotesize{2015}}{Semantic Cloud Platform (SCP)}{}{}{
[\href{https://github.com/Adoni/WordPlatform-Beiyu}{\textcolor{color1}{Code}}][\href{http://103.37.150.140}{\textcolor{color1}{Demo}}]
}{
\begin{itemize}
    \item {Supported by Beijing Language and Culture University. The main purpose was to detect semantic vicissitude.}
    \item {Using word2vec to learn word embeddings in each year and tooking the nearest neighbors as the possible semantics, which was clustered by Affinity Propagation a shared embedding space. The final clustering results were regarded as the semantics of each word.}
\end{itemize}}
\cventry{\footnotesize{2015}}{User Profiling on E-commerce Websites from Online Review}{}{}{}{
